jQuery is a fast, small, and feature-rich JavaScript library. It makes things like HTML document traversal and manipulation, event handling, animation, and Ajax much simpler with an easy-to-use API that works across a multitude of browsers. With a combination of versatility and extensibility, jQuery has changed the way that millions of people write JavaScript.

In this tutorial, we'll build a simple html page which uses jQuery to gather values from traditional form inputs, process them, and write them back to the page.

Add jQuery — Submit function

Next, let's add the function that we call on clicking our Submit button.

function outputForm() {

   // Get the value of the simple text input field
   var name = $('#name').val();

   // Get the value of the dropdown
   var identifyWith = $('#identify-with').val();

   // Get the values of the checkboxes
   var weaponOfChoice = [];
      $.each($("input[name='weapon']:checked"), function(){
          weaponOfChoice.push( $(this).val() );
      });

   var allWeapons = weaponOfChoice.join(", ") + ".";

   // Get the value of the chosen radio button
   var whichPlanet = $("input[name='planets']:checked").val();

   // Get the value of the textarea
   var moreInfo = $.trim( $("#more-about-me").val() );

   // Concatenate a string from all the variables
   outputString = "My name is " + name + " and I'm " + identifyWith + " My weapon(s) of choice is/are: " + allWeapons + " I come from the planet " + whichPlanet + ". " + moreInfo;

   // Write the string to the output div
   $("#profile-box").html("<p class='lead'>" + outputString + "</p>");

};
